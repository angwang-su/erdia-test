"categories(Category)" {
  *number id    PK    
  *varchar name    UK    
  varchar description    
  *integer postCount    
  *timestamp createdAt    
  *timestamp updatedAt    
}


"categories(Category)"  }o  --  o{  "posts(Post)":  "post_id,post_id"

"comments(Comment)" {
  *number id    PK    
  *text content    
  *integer likeCount    
  *boolean isDeleted    
  *timestamp createdAt    
  *timestamp updatedAt    
  number author    FK    
  number post    FK    
  number parent    FK    
}


"comments(Comment)"  }o  --  o|  "users(User)":  "authorId"
"comments(Comment)"  }o  --  o|  "posts(Post)":  "postId"
"comments(Comment)"  }o  --  o|  "comments(Comment)":  "parentId"

"likes(Like)"  }o  --  o|  "users(User)":  "userId"
"likes(Like)"  }o  --  o|  "posts(Post)":  "postId"
